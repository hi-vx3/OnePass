JavaScript مثال 

// مصادقة المستخدم باستخدام BrandMe API
fetch("https://api.brandme.me/auth", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": "Bearer YOUR_CLIENT_SECRET"
  },
  body: JSON.stringify({ code: "USER_AUTH_CODE" })
})
.then(res => res.json())
.then(data => {
  console.log("معلومات المستخدم:", data);
  // {
  //   "id": "user_123",
  //   "email": "user@brandme.me",
  //   "name": "اسم المستخدم",
  //   "avatar": "https://..."
  // }
})
.catch(error => {
  console.error("خطأ في المصادقة:", error);
});
