أمثلة دمج BrandMe Login

تعرف على كيفية إضافة زر تسجيل الدخول وحماية البريد الوهمي لمستخدميك.

اختر لغة أو إطار العمل

مثال JavaScript - مصادقة المستخدم

POST
https://api.brandme.me/auth

هذا الكود يقوم بمصادقة المستخدم باستخدام BrandMe API واسترجاع معلوماته. 


// مصادقة المستخدم باستخدام BrandMe API
fetch("https://api.brandme.me/auth", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": "Bearer YOUR_CLIENT_SECRET"
  },
  body: JSON.stringify({ 
    code: "USER_AUTH_CODE",
    client_id: "YOUR_CLIENT_ID"
  })
})
.then(response => {
  if (!response.ok) {
    throw new Error('Network response was not ok');
  }
  return response.json();
})
.then(data => {
  console.log("معلومات المستخدم:", data);
  // {
  //   "user_id": "user_123",
  //   "email": "user@brandme.me",
  //   "name": "اسم المستخدم",
  //   "avatar": "https://example.com/avatar.png"
  // }
})
.catch(error => {
  console.error("حدث خطأ في المصادقة:", error);
});

مثال JavaScript - إضافة زر التسجيل

Frontend

هذا الكود يوضح كيفية إضافة زر تسجيل الدخول إلى واجهة المستخدم. 


// إضافة زر تسجيل الدخول إلى الصفحة
function addBrandMeLoginButton() {
  const button = document.createElement('button');
  button.innerHTML = `
    <div style="display: flex; align-items: center; justify-content: center;">
      <img src="brandme-logo.svg" alt="BrandMe" style="width: 20px; margin-left: 8px;">
      تسجيل الدخول مع BrandMe
    </div>
  `;
  button.style = `
    background-color: #2563EB;
    color: white;
    padding: 12px 24px;
    border: none;
    border-radius: 8px;
    cursor: pointer;
    font-weight: 500;
    margin: 16px 0;
  `;
  button.onclick = () => {
    // توجيه المستخدم إلى صفحة مصادقة BrandMe
    window.location.href = 'https://brandme.me/auth?client_id=YOUR_CLIENT_ID';
  };
  document.body.appendChild(button);
}

// استدعاء الدالة عند تحميل الصفحة
document.addEventListener('DOMContentLoaded', addBrandMeLoginButton);
